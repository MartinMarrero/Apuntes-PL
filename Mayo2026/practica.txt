2. a. Is the grammar ambiguous? If so, propose an unambiguous one.

Yes, the grammar is ambiguous, this is caused by the lack of precedences in the grammar;
for example, for the following line of code:

a + b * c

the compiler can read:

(a + b) * c

or also:

a + (b * c)

a proposal for a non-ambiguous grammar is the following:

program -> decls ';' assign_expr
        | assign_expr

decls -> decl ';' decls
    	| decl

decl -> type id

type -> int
			| float

assign_expr -> id '=' add_expr
						| add_expr

add_expr -> add_expr '+' mult_expr
					| add_expr '-' mult_expr
					| mult_expr

mult_expr -> mult_expr '*' unary_expr
					| mult_expr '/' unary_expr
					| unary_expr

unary_expr -> '-' primary
						| primary

primary -> num
				| id
				| '(' assign_expr ')'
        | '{' decls ';' assign_expr '}'
        | '{' assign_expr '}'

			SymbolEntry {
				name: string
				type: INT | FLOAT
				kind: VAR
			}

			SymbolTable = stack<HashTable<name, SymbolEntry>>
			
			
			with operations:
				enter_scope() // push to the stack
				exit_scope()  // pop from the stack
				insert(name, entry)
				lookup(name)

			program -> decls ';' assign_expr
			{ 
				enter_scope();
				decls.env = env;
				assign_expr.env = env;
				exit_scope();
			}

			program -> assign_expr
			{
				enter_scope();
				assign_expr.env = env;
				exit_scope();
			}

			decls -> decl ';' decls2
			{
				decl.env = env;
				decls2.env = env;
			}

			decls -> decl
			{
				decl.env = env;
			}

			decl -> type id;

				if(lookup(id) != null) { error("variable redeclaration"); }
				insert(ID, SymbolEntry(ID, type.type));
			}

			}

			}

			}

			T ::= INT | FLOAT

			type(ID) = lookup(id).type

			type(e1 + e2) =

					error("Incorrect type");

			(same for - and *)

			type(e1 / e2) =
				if(type(e1) == ERROR || type(e2) == ERROR) { error("Incorrect type"); }
				else { return FLOAT; }

			type(id = e1) =
				if(type(id) == ERROR || type(e1) == ERROR) { error("Incorrect type"); }
				elsif(type(id) == FLOAT || type(e1) == FLOAT) { return FLOAT; }
				else { return INT; }
